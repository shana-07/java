  import java.util.Scanner;
   class RightTriangle{
       public static void main(String[]args){
           System.out.println("right angled triangle:");
           for(int i = 1; i <= 5; i++){
              for(int j = 1; j <= i;  j++){
                  System.out.print("*");
               }
                   System.out.println();
          }
      }
  }
